Another way of Representing Trees

You will sometimes see code for trees done somewhat less recursively (the code is still recursive but the definition of the tree is less so):

public class BinarySearchTree {

private TreeNode root;

//***********************************************************************

private class TreeNode {

private Comparable key;

private TreeNode left;

private TreeNode right;

private TreeNode (Comparable k) {

key=k;

}

}

//************************************************************************

//instead of using insert we let it call a recursive version – insertKey

//this code is more or less from Standish

public void insert (Comparable k) {

root=insertKey(root ,k);

}

private TreeNode insertKey(TreeNode t, Comparable k) {

if (t==null) {

TreeNode temp=new TreeNode(k);

return temp;

}

else if (k.compareTo(t.key) <0) {

t.left=insertKey(t.left, k);

return t;

}

else if (k.compareTo(t.key) >0) {

t.right=insertKey(t.right, k);

return t;

}

else //equal – a duplicate – what do you want to do??

}

//**************************************************************************

//in the code below we go one further and don’t use recursion

public TreeNode find(Comparable k) {

TreeNode T = root;

while (T != null) {

if (k.compareTo(T.key)) < 0) {

T = T.left;

}

else if (k.compareTo(T.key)==0) {

return T;

}

else {

T = T.right;

}

}

return T; // returns null, if search failed

}
